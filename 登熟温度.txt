CREATE OR REPLACE VIEW average_temp AS SELECT to_days(date) 日付, avg(temp) 平均気温 FROM atmos GROUP BY to_days(date);
CREATE OR REPLACE VIEW before_temp AS SELECT A.日付, A.平均気温, B.平均気温 C FROM average_temp A, average_temp B WHERE B.日付 <= A.日付;
SELECT 日付, 平均気温, sum(C) as 登熟温度 FROM before_temp WHERE 日付 >= '2022-08-08' AND C >= '2022-08-08' GROUP BY 日付;
